# 十、编辑个人信息

# 准备

创建文件并写入以下内容：

<template>
  <div>
    <van-nav-bar
      title="个人信息"
      left-text="返回"
      left-arrow
      right-text="保存"
      @click-left="$router.back()"
      @click-right="onSave"
    />
    <van-cell-group>
      <van-cell title="头像" is-link>
        <van-image
          round
          width="30"
          height="30"
          :src="userProfile.photo"
        />
      </van-cell>
      <van-cell title="昵称" :value="userProfile.name" is-link />
      <van-cell title="性别" :value="userProfile.gender === 0 ? '男' : '女'" is-link />
      <van-cell title="生日" :value="userProfile.birthday" is-link />
    </van-cell-group>
  </div>
</template>

<script>
import { getUserProfile } from '@/api/user'

export default {
  name: 'UserEdit',
  data () {
    return {
      userProfile: {}
    }
  },

  created () {
    this.loadUserProfile()
  },

  methods: {
    async loadUserProfile () {
      const { data } = await getUserProfile()
      this.userProfile = data.data
    },

    onSave () {
      console.log('提交表单')
    }
  }
}
</script>

配置路由：

import Vue from 'vue'
import Router from 'vue-router'
import Login from '@/views/login'
import Home from '@/views/home'
import Tabbar from '@/views/tabbar'
import Search from '@/views/search'
import SearchResult from '@/views/search-result'
import Article from '@/views/article'
import User from '@/views/user'
import UserEdit from '@/views/user-edit'

Vue.use(Router)

export default new Router({
  routes: [
    {
      path: '/',
      component: Tabbar,
      children: [
        {
          name: 'home',
          path: '',
          component: Home
        },
        {
          name: 'user',
          path: '/user',
          component: User
        }
      ]
    },
    {
      name: 'login',
      path: '/login',
      component: Login
    },
    {
      name: 'search',
      path: '/search',
      component: Search
    },
    {
      name: 'search-result',
      path: '/search/:q',
      component: SearchResult
    },
    {
      name: 'article',
      path: '/article/:id',
      component: Article
    },
    {
      name: 'user-edit',
      path: '/user-edit',
      component: UserEdit
    }
  ]
})

  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">

<body>
  <div>
    <!-- 
      src 文件路径
      图片能以字符串的方式存储在网页中，这个字符串叫做 base64 编码格式的字符串
      
      http 网络文路径
      本地文件路径

        结合后端处理
          后端提供接口，请求把图片传到后端，后端返回一个网络路径，将收到的这个网络路径作为 img 的 src
        纯前端处理
          base64 编码字符串
          JavaScript 有一个API可以把 file 类型的 input 所选择的文件转为 base64 格式字符串

图片转 base64：

function getBase64(file) {
   var reader = new FileReader();
   reader.readAsDataURL(file);
   reader.onload = function () {
     console.log(reader.result);
   };
   reader.onerror = function (error) {
     console.log('Error: ', error);
   };
}

var file = document.querySelector('#files > input[type="file"]').files[0];
getBase64(file); // prints the base64 string

图片转 base64（Promise）：

function getBase64(file) {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.readAsDataURL(file);
    reader.onload = () => resolve(reader.result);
    reader.onerror = error => reject(error);
  });
}

var file = document.querySelector('#files > input[type="file"]').files[0];
getBase64(file).then(
  data => console.log(data)
);

createObjectURL 和 FileReader 的区别：

执行方式：

  • createObjectURL 是同步的
  • FileReader 是异步的

内存使用：

  • createObjectURL 读取返回一个 Blog 对象， 然后将读取结果存储在内存中，直到 document 触发 unload 事件，例如关闭文档或者执行 revokeObjectURL 才会被释放
  • FileReader 直接返回 base64 格式的字符串，相比 Blog 对象的内存占用要更高一些，但是好在它会被自动垃圾回收

兼容性：

  • createObjectURL 兼容 IE10 和更现代的浏览器
  • FileReader 兼容 IE10 和更现代的浏览器

个人建议使用 FileReader。
