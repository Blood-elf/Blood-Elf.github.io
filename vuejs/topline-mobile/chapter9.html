# 展示用户信息

在 api/user.js 中新增：

/**
 * 获取用户
 */
export const getUser = userId => {
  return request({
    method: 'GET',
    url: '/app/v1_0/user',
    data: {
      target: userId
    }
  })
}

...

  ...

    return {
      userInfo: {}

    this.loadUser()
  },

  methods: {
    async loadUser () {
      const { data } = await getSelf()
