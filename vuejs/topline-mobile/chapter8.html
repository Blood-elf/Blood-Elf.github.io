</script>

    round

  >
    <!-- 顶部导航栏 -->

    <!-- 当前评论 -->

    <!-- /当前评论 -->

    <!-- 回复列表 -->
    <van-list
      v-model="loading"
      :finished="finished"
      finished-text="没有更多了"
      @load="onLoad"
    >
      <van-cell
        v-for="comment in comments"
        :key="comment.com_id.toString()"
      >
        <van-image
          slot="icon"
          round
          width="30"
          height="30"
          style="margin-right: 10px;"
          :src="comment.aut_photo"
        />
        <span style="color: #466b9d;" slot="title">{{ comment.aut_name }}</span>
        <div slot="label">
          <p style="color: #363636;">{{ comment.content }}</p>
          <p>
            <span style="margin-right: 10px;">{{ comment.pubdate | relativeTime }}</span>
            <van-button size="mini" type="default">回复</van-button>
          </p>
        </div>
        <van-icon slot="right-icon" name="like-o" />
      </van-cell>
    </van-list>
    <!-- /回复列表 -->

    <!-- 添加回复 -->
    <van-cell-group class="publish-wrap">
      <van-field
        clearable
        placeholder="请输入评论内容"
        v-model="commentInput"
      >
        <van-button
          slot="button"
          size="mini"
          type="info"
          @click="onPublishComment"
        >发布</van-button>
      </van-field>
    </van-cell-group>
    <!-- /添加回复 -->
  </van-popup>
</template>

<script>
import { getComments, addComment } from '@/api/comment'

export default {
  name: 'ArticleComment',
  data () {
    return {
      show: true,
      comments: [],
      loading: false,
      finished: false,
      offset: null,
      commentInput: ''
    }
  },

  computed: {
    articleId () {
      return this.$route.params.articleId
    }
  },

  created () {
  },

  methods: {
    async onLoad () {
      const { data } = await getComments({
        type: 'a',
        source: this.articleId,
        offset: this.offset
      })

      const { results, last_id: lastId } = data.data

      this.comments.push(...results)

      if (lastId) {
        // 设置请求下一页的标记
        this.offset = lastId
      } else {
        // 所有数据加载结束
        this.finished = true
      }

      // 取消本次的加载更多的 loading，才能开始下一次
      this.loading = false
    },

    async onPublishComment () {
      // 拿到文本框的内容
      const commentInput = this.commentInput.trim()

      // 非空校验
      if (!commentInput.length) {
        return
      }

      const { data } = await addComment({
        target: this.articleId,
        content: commentInput
      })

      // 将频道数据放到列表顶部，给用户一个反馈展示
      this.comments.unshift(data.data.new_obj)

      // 清空文本框
      this.commentInput = ''

      this.$toast('发布成功')
    }
  }
}
</script>
